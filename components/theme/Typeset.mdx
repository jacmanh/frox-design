export const fontSizes = Object.entries(commonTheme.font.size)
  .filter(([key]) => key !== 'default' && key !== 'mobile' && key !== 'header')
  .map(([, value]) => {
    return Number(value.slice(0, -3)) * 16 + 'px'
  })
